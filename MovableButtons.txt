// ·:·:· Movable ability buttons ·:·:· //
//
// Each ability has a button with an "o" that
// is its HANDLE: drag the "o" to move the
// button; tap the button body to activate the
// ability.
//
// The potion button is always visible (except
// when empty/used). Ability buttons are hidden
// and only appear when the matching item is
// equipped.
//
// Positions are remembered (posX/posY persist
// across Ouroboros loops). Default position is
// shared per side, so every R ability stacks
// on the same spot (the mask spot); potion and
// L abilities get their own column, so a potion
// + one R + one L can be seen at once.
//
// Lists keep the code short. There is ONE
// function (OnTap): it is required because a
// Button swallows the tap and the only
// supported way to react to a press is the
// SetPressed callback. Dragging the "o" (a
// Text) works without functions.
//
// To add abilities: add entries to 'ids',
// 'sides' and 'labels'.
//   side = R, L, RL  (where it is equipped)
//   side = P         (potion, always visible)

disable hud

// --- Button data (lists) ---
// Blade, Shovel, Heavy Hammer, Bardiche,
// Skeleton Arm and the Ice (Eternity) Staff are
// equipped with "equip" only (not equipR/L),
// but their ability activates with R, so they
// all use side "R" just like the mask.
var ids    = ["potion", "mask", "blade", "shovel", "heavy_hammer", "bardiche", "skeleton_arm", "quarterstaff", "hatchet", "staff_ice", "staff_fire", "staff_poison", "staff_vigor", "staff_stone", "staff_aether"]
var sides  = ["P",      "R",    "R",     "R",      "R",            "R",        "R",            "R",            "R",        "R",         "R",          "R",            "R",           "R",           "R"]
var labels = ["POTION", "MASK", "BLADE", "SHOVEL", "HAMMER",       "BARDICHE", "SKELARM",      "QUARTER",      "HATCHET",  "ICESTAFF",  "FIRESTAFF",  "PSNSTAFF",     "VIGORSTAFF",  "STONESTAFF",  "AETHRSTAFF"]

// --- Persistent positions (survive loops) ---
var posX = []
var posY = []

// --- Internal state ---
var buttons = []   // one button per entry
var handles = []   // one "o" (handle) per entry
var sel = -1       // button being dragged (-1 = none)
var offX = 0
var offY = 0
var o              // loop helpers
var bt
var eq = false

// --- The only function: activate on tap ---
func OnTap(btn)
 var i = buttons.IndexOf(btn)
 ?i < 0
  return
 ?sides[i] = P
  activate potion
 :?sides[i] = L
  activate L
 :?sides[i] = R
  activate R
 :
  // RL: activate whichever hand holds it
  ?item.left = ids[i]
   activate L
  :
   activate R

// --- Default positions (run once, then kept) ---
?posX.Count() = 0
 for i = 0 .. ids.Count() - 1
  posY.Add(2)
  ?sides[i] = P
   posX.Add(2)    // potion column
  :?sides[i] = L
   posX.Add(20)   // L column
  :
   posX.Add(11)   // R column (mask spot)

// --- Build everything in a single loop ---
?loc.begin | loc.loop
 ui.Clear()
 buttons.Clear()
 handles.Clear()
 sel = -1
 for i = 0 .. ids.Count() - 1
  // the "o" handle
  o = ui.AddText()
  o.anchor = top_left
  o.dock = top_left
  o.text = "o"
  o.w = 3
  o.h = 1
  o.x = posX[i]
  o.y = posY[i]
  o.visible = false
  handles.Add(o)
  // the button, right below its "o"
  bt = ui.AddButton()
  bt.anchor = top_left
  bt.dock = top_left
  bt.text = labels[i]
  bt.w = 7
  bt.h = 3
  bt.x = posX[i]
  bt.y = posY[i] + 1
  bt.visible = false
  bt.SetPressed(OnTap)
  buttons.Add(bt)

// --- Show/hide every frame ---
for i = 0 .. ids.Count() - 1
 eq = false
 ?sides[i] = P
  // potion: visible unless empty
  ?item.potion ! empty
   eq = true
 :
  // equipped? match by Appendix ID. Staffs need
  // armor to cast, so they appear via
  // CanActivate (which requires armor). No
  // cooldown check, so the button hides the
  // instant you switch weapons.
  ?item.right = ids[i] | item.left = ids[i] |
  ^ item.CanActivate(ids[i])
   eq = true
 buttons[i].visible = eq
 handles[i].visible = eq

// --- Grab: drag by the "o" (visible only) ---
?key = primaryBegin
 for i = 0 .. handles.Count() - 1
  o = handles[i]
  ?o.visible = true & input.x >= o.x &
  ^ input.x < o.x + o.w & input.y >= o.y &
  ^ input.y < o.y + o.h
   sel = i
   offX = input.x - o.x
   offY = input.y - o.y
   ui.root.Add(buttons[sel])
   ui.root.Add(o)

// --- Drag: the "o" follows the cursor ---
?key = primary & sel >= 0
 o = handles[sel]
 bt = buttons[sel]
 o.x = input.x - offX
 o.y = input.y - offY
 bt.x = o.x
 bt.y = o.y + 1
 posX.Emplace(sel, o.x)   // remember position
 posY.Emplace(sel, o.y)

// --- Release ---
?key = primaryEnd
 sel = -1
