// ·:·:· Movable ability buttons ·:·:· //
//
// Each ability has a button with an "o" that
// is its HANDLE: drag the "o" to move the
// button; tap the button body to activate the
// ability.
//
// The potion button is always visible (except
// when empty/used). Ability buttons are hidden
// and only appear when the matching item is
// equipped.
//
// Positions are remembered (posX/posY persist
// across Ouroboros loops). Default position is
// shared per side, so every R ability stacks
// on the same spot

disable hud a

var arts = [potionArt, maskArt, bladeArt, shovelArt, hammerArt, bardicheArt, skelArt, quarterArt, hatchetArt, iceStaffArt, fireStaffArt, poisonStaffArt, vigorStaffArt, stoneStaffArt, aetherStaffArt, lollipoArt, icewandArt, firewandArt, poisonwandArt, vigorwandArt, stonewandArt, etherwandArt, firetalisArt, ethertalisArt, cinderArt, voidArt]

// --- Persistent positions (survive loops) ---
var posX = []
var posY = []

// --- Internal state ---
var buttons = []   // one button per entry
var handles = []   // one "o" (handle) per entry
var artels  = []   // one ascii-art overlay per entry
var sel = -1       // button being dragged (-1 = none)
var offX = 0
var offY = 0
var o              // loop helpers
var bt
var art
var eq = false

func OnTap(btn)
 var i = buttons.IndexOf(btn)
 ?i < 0
  return
 ?sides[i] = P
  activate potion
 :?sides[i] = SUM
  // summon devour (cinderwisp/voidweaver)
  activate @ids[i]@
 :?sides[i] = L
  activate L
 :?sides[i] = R
  activate R
 :?sides[i] = RLT
  // talisman: re-equip its hand, then activate, so
  // the talisman is cast (not the active summon)
  ?item.left = ids[i]
   equipL @ids[i]@
   activate L
  :
   equipR @ids[i]@
   activate R
 :
  // RL (wands, etc.): activate the hand holding it
  ?item.left = ids[i]
   activate L
  :
   activate R

?posX.Count() = 0
 for i = 0 .. ids.Count() - 1
  posY.Add(2)
  ?sides[i] = P
   posX.Add(2)    
  :?sides[i] = L
   posX.Add(20)  
  :?sides[i] = R
   posX.Add(11)  
  :?sides[i] = SUM
   posX.Add(47)
  :
   posX.Add(38)

// --- Build everything in a single loop ---
?loc.begin | loc.loop
 ui.Clear()
 buttons.Clear()
 handles.Clear()
 artels.Clear()
 sel = -1
 for i = 0 .. ids.Count() - 1
  // the "o" handle
  o = ui.AddText()
  o.anchor = top_left
  o.dock = top_left
  o.text = "o"
  o.w = 3
  o.h = 1
  o.x = posX[i]
  o.y = posY[i]
  o.visible = false
  handles.Add(o)
  bt = ui.AddButton()
  bt.anchor = top_left
  bt.dock = top_left
  bt.text = ""
  bt.w = 6
  bt.h = 4
  bt.x = posX[i]
  bt.y = posY[i] + 1
  bt.visible = false
  bt.SetPressed(OnTap)
  buttons.Add(bt)
  art = ui.AddText()
  art.anchor = top_left
  art.dock = top_left
  art.align = left
  art.w = 7
  art.h = 3
  art.x = posX[i] + 1
  art.y = posY[i] + 2
  art.visible = false
  art.text = arts[i]
  artels.Add(art)

// --- Show/hide every frame ---
for i = 0 .. ids.Count() - 1
 eq = false
 ?sides[i] = P
  // potion: visible unless empty
  ?item.potion ! empty
   eq = true
 :?sides[i] = SUM
  // summon button: visible while it is summoned.
  // Cinderwisp is special: its button only shows
  // once it has attacked a foe (ignition var > 0).
  ?summon.GetId() = ids[i]
   ?ids[i] = "cinderwisp"
    ?summon.GetVar("ignition") > 0
     eq = true
   :
    eq = true
 :
  ?item.right = ids[i] | item.left = ids[i] |
  ^ item.CanActivate(ids[i])
   eq = true
 ?(sides[i] = RL | sides[i] = RLT) & eq
  ?item.left = ids[i]
   handles[i].x = 29
   buttons[i].x = 29
   artels[i].x = 30
  :
   handles[i].x = 38
   buttons[i].x = 38
   artels[i].x = 39
 buttons[i].visible = eq
 handles[i].visible = eq
 artels[i].visible = eq

// --- Grab: drag by the "o" (visible only) ---
?key = primaryBegin
 for i = 0 .. handles.Count() - 1
  o = handles[i]
  ?o.visible = true & input.x >= o.x &
  ^ input.x < o.x + o.w & input.y >= o.y &
  ^ input.y < o.y + o.h
   sel = i
   offX = input.x - o.x
   offY = input.y - o.y
   ui.root.Add(buttons[sel])
   ui.root.Add(artels[sel])
   ui.root.Add(o)

// --- Drag: the "o" follows the cursor ---
?key = primary & sel >= 0
 o = handles[sel]
 bt = buttons[sel]
 art = artels[sel]
 o.x = input.x - offX
 o.y = input.y - offY
 bt.x = o.x
 bt.y = o.y + 1
 art.x = o.x + 1
 art.y = o.y + 2
 posX.Emplace(sel, o.x)   // remember position
 posY.Emplace(sel, o.y)

// --- Release ---
?key = primaryEnd
 sel = -1
