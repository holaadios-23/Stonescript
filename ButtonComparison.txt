// --- 2. Función Helper para crear botones ---
// Esta función encapsula el estilo para que ambos métodos se vean igual.
func MyButtonHelper(txt, px, py, act)
  var b = ui.AddButton()
  b.text = txt
  b.x = px
  b.y = py
  b.w = 15
  b.h = 3
  b.SetPressed(act)
  return b

// --- 3. Bucle de Inicialización ---

?loc.begin | loc.loop
  ui.Clear()
  ui_init = false

?!ui_init
  // --- MÉTODO A: USANDO FUNCIONES (Manual) ---
  // Aquí llamamos a la función por cada botón que queremos.
  // Es rígido: si quieres 10 botones, escribes 10 líneas.
  MyButtonHelper("Boton F1", -15, 2, AccionManual)
  MyButtonHelper("Boton F2", -15, 6, AccionManual)

  // --- MÉTODO B: USANDO LISTAS (Arrays) ---
  // Aquí definimos los datos primero y luego iteramos.
  // Es flexible: si añades un elemento al array, el botón se crea solo.
