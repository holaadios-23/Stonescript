// CruzRojaCounter.txt
// Gestiona el estado de la "Cruz Roja" basado en el balance neto de HP (Wiki Logic)

var secure = true
var loop = 0
var hp_loss = 0
var hp_gain = 0
var last_hp = 0

?loc.begin
  loop = 0
  secure = true
  hp_loss = 0
  hp_gain = 0
  last_hp = hp

// --- Monitorización de balance de vida (Lógica de Cruz Roja) ---
var current_hp
current_hp = hp

?current_hp < last_hp
  var d
  d = last_hp - current_hp
  hp_loss = hp_loss + d
:?current_hp > last_hp
  var g
  g = current_hp - last_hp
  // Registramos curación (lifesteal, vigor, etc.)
  hp_gain = hp_gain + g

// Si el daño total supera a la sanación acumulada, la run se marca como fallida
?hp_loss > hp_gain
  secure = false
  loop = 0

last_hp = current_hp

// --- Lógica al completar un Loop ---
?loc.loop
  ?secure
    loop++
  :
    loop = 0
  
  // Resetear balance para la siguiente vuelta de Ouroboros
  secure = true
  hp_loss = 0
  hp_gain = 0
  last_hp = hp

// --- Visualización de Mensajes ---
?loop >= 20
  >`0,10,#green,cruz roja quitada
:?secure = false | loop = 0
  >`0,10,#red,alerta cruz roja (@loop@/20)
:
  >`0,10,#yellow,Progreso limpio: @loop@/20
