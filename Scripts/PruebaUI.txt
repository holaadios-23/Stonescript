// Map item IDs to their max cooldown in frames (inspired by ClickAbilities.txt)
func GetMaxCD(id)
 ?id = "bash"
  return 270
 ?id = "dash"
  return 45
 ?id = "quarterstaff"
  return 195
 ?id = "staff_fire"
  return 900
 ?id = "mind"
  return 360
 ?id = "blade"
  return 2400
 return 0

// Stats variables
var fps = 0
var dps = 0
var hps = 0
var fps_c = 0
var sec_l = 0
var hp_l = 0
var dmg_l = []

// Updates FPS, DPS, and HPS every frame
func UpdateStats()
 fps_c++
 ?time.second ! sec_l
  fps = fps_c
  fps_c = 0
  sec_l = time.second
 var d = 0
 var hp_n = foe.hp + foe.armor
 ?hp_n < hp_l
  d = hp_l - hp_n
 hp_l = hp_n
 dmg_l.Add(d)
 ?dmg_l.Count() > 30
  dmg_l.RemoveAt(0)
 dps = 0
 hps = 0
 for x : dmg_l
  dps += x
  ?x > 0
   hps++


// Helper to create text elements
func createtext(texto, x, y, w, h, color)
 var t = ui.AddText()

 t.color = color
 t.text = texto
 return t

// Helper to create buttons
func CrearBoton(texto, w, h, accion)
  var b = ui.AddButton()
  b.text = texto
  b.w = w
  b.h = h
  b.anchor = bottom_center
  b.dock = top_left 
  b.SetPressed(accion)
  return b

// Returns elemental info and weakness for the foe
func GetElementalInfo(id)
  var element = "o"
  var weakness = "o"
  ?id = ice
    element = "❄"
    weakness = "φ"
  :?id = poison
    element = "∞"
    weakness = "❄"
  :?id = vigor
    element = "♥"
    weakness = "∞"
  :?id = aether
    element = "*"
    weakness = "♥"
  :?id = fire
    element = "φ"
    weakness = "*"
  
  ?!id
    return "Elements: none"
      
  return "Foe: " + element + " weak to " + weakness


// Returns a safe string for IDs (avoids nulls)
func SafeName(id)
 ?id
  return id
 return "none"

// Calculates max GP based on location stars
func MaxGP()
 var gp = 999999
 ?loc.stars = 5
  gp = 150
 ?loc.stars = 10
  gp = 200
 ?loc = "rocky"
  gp = gp * 3
 return gp

// Generates cooldown text with color-coded percentages (Red/Yellow/Green)
func GetCDText(list)
 var s = ""
 for id : list
  var cd_curr = item.GetCooldown(id)
  var cd_max = GetMaxCD(id)
  
  // If item is not in map, skip to avoid errors
  ?cd_max = 0
   continue

  var percent = math.Floor(math.Min(100, (cd_max - cd_curr) / cd_max * 100))

  s += id + ": " + percent + "%#\n"
 return s
